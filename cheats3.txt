    function getTitle(statName) {
        var title = statName.replace(/[^a-zA-Z0-9 ]/g, " ");
        return title.charAt(0).toUpperCase() + title.slice(1);

    // Parse stat charts from choicescript_stats.txt
    function parseStatChart(rawStatChart) {
        var statIndex = 0;
        var statCharts = [];
        
        for (var i = 0; i < rawStatChart.length; i++) {
            var line = rawStatChart[i];
            var pieces = line.trim().split(' ');
            var type = pieces[0];
            var variable = pieces[1];
            var label = '';
            
            if (type == 'text' || type == 'percent') {
                if (pieces.length > 2) {
                    label = line.substring(line.indexOf(pieces[2])).trim();
                } else {
                    label = variable;
                }
                statCharts.push({ statIndex: statIndex, type: type, variable: variable.toLowerCase(), label: label });
                statIndex++;
            } else if (type == 'opposed_pair') {
                var nextLine = rawStatChart[i + 1];
                var opposed_label = nextLine ? nextLine.trim() : '';
                statCharts.push({ 
                    statIndex: statIndex, 
                    type: type, 
                    variable: variable.toLowerCase(), 
                    label: variable,
                    opposed_label: opposed_label 
                });
                statIndex++;
                i++;
            }
        }
        return statCharts;
    }
    
    // Load stat charts
    function loadStatCharts() {
        return new Promise(function(resolve) {
            var baseURL = getBaseURL();
            var statsURL = baseURL + "mygame/scenes/choicescript_stats.txt";
            
            console.log('Loading choicescript_stats.txt...');
            
            fetch(statsURL)
                .then(function(response) {
                    if (!response.ok) throw new Error("Network response was not ok");
                    return response.text();
                })
                .then(function(text) {
                    var lines = text.split(/\r?\n/);
                    var currentLine = 0;
                    var allStatCharts = [];
                    
                    while (currentLine < lines.length) {
                        var line = lines[currentLine];
                        if (line.trim().startsWith('*stat_chart')) {
                            var rawStatChart = [];
                            var firstIndent = line.search(/\S|$/);
                            currentLine++;
                            while (currentLine < lines.length) {
                                line = lines[currentLine];
                                var skippable = line.trim().length == 0 || line.trim().startsWith('*comment');
                                if (skippable) {
                                    currentLine++;
                                } else {
                                    var currentIndent = line.search(/\S|$/);
                                    if (currentIndent > firstIndent) {
                                        rawStatChart.push(line);
                                        currentLine++;
                                    } else {
                                        break;
                                    }
                                }
                            }
                            var parsed = parseStatChart(rawStatChart);
                            allStatCharts = allStatCharts.concat(parsed);
                        } else {
                            currentLine++;
                        }
                    }
                    
                    window.cheatData.statCharts = allStatCharts;
                    console.log('Stat charts loaded:', allStatCharts.length);
                    resolve(allStatCharts);
                })
                .catch(function(error) {
                    console.error("Error loading stat charts:", error);
                    resolve([]);
                });
        });
    }
    
    // Compile stats
    function compileStats() {
        var stats = window.stats;
        if (!stats) {
            console.error('Stats object not found');
            return;
        }
        
        var statKeyValues = Object.entries(stats);
        var modifiableStatChartsStats = [];
        var modifiableNumericalStats = [];
        var modifiableBooleanStats = [];
        var modifiableStringStats = [];
        
        // Compile stat charts
        for (var i = 0; i < window.cheatData.statCharts.length; i++) {
            var stat = window.cheatData.statCharts[i];
            var value = stats[stat.variable];
            if (value !== undefined) {
                var val = value;
                var statType = typeof value;
                if (statType === 'boolean') {
                    modifiableStatChartsStats.push({
                        index: stat.statIndex,
                        key: stat.variable,
                        value: val,
                        statType: 'boolean',
                        displayType: stat.type,
                        label: stat.label,
                        opposedLabel: stat.opposed_label || ''
                    });
                } else if (!isNaN(value) && value !== null && value !== '') {
                    modifiableStatChartsStats.push({
                        index: stat.statIndex,
                        key: stat.variable,
                        value: parseInt(value),
                        statType: 'number',
                        displayType: stat.type,
                        label: stat.label,
                        opposedLabel: stat.opposed_label || ''
                    });
                } else if (typeof val === 'string') {
                    modifiableStatChartsStats.push({
                        index: stat.statIndex,
                        key: stat.variable,
                        value: val,
                        statType: 'string',
                        displayType: stat.type,
                        label: stat.label,
                        opposedLabel: stat.opposed_label || ''
                    });
                }
            }
        }
        
        // Compile all stats
        for (var key in stats) {
            var value = stats[key];
            if (typeof value === 'boolean') {
                modifiableBooleanStats.push({ key: key, value: value, displayType: 'boolean' });
            } else if (!isNaN(value) && value !== null && value !== '' && typeof value !== 'boolean') {
                modifiableNumericalStats.push({ key: key, value: parseInt(value), displayType: 'number' });
            } else if (typeof value === 'string') {
                modifiableStringStats.push({ key: key, value: value, displayType: 'string' });
            }
        }
        
        window.cheatData.modifiableStatChartsStats = modifiableStatChartsStats;
        window.cheatData.modifiableNumericalStats = modifiableNumericalStats;
        window.cheatData.modifiableBooleanStats = modifiableBooleanStats;
        window.cheatData.modifiableStringStats = modifiableStringStats;
    }

    // Open cheat window
    function openCheatWindow() {
        console.log('Opening cheat window...');
        
        // Compile stats first
        compileStats();
        
        // Build HTML
        var html = buildCheatHTML();
        
        // Open window
        var childWindow = window.open('', 'cheatWindow', 'width=900,height=700,toolbar=0,menubar=0,location=0,status=0,scrollbars=1,resizeable=1');
        if (!childWindow) {
            alert('Please allow popups for this site to use the cheat tool.');
            return;
        }
        
        childWindow.document.title = 'ChoiceScript Cheat Editor';
        
        // Add the HTML
        childWindow.document.body.innerHTML = html;
        
        // Add functions to child window
        childWindow.cheatData = window.cheatData;
        childWindow.stats = window.stats;
        
        // Define helper functions in child window
        childWindow.cheatShowTab = function(tabIndex) {
            var tabs = childWindow.document.querySelectorAll('.tabContentItem');
            tabs.forEach(function(tab, index) {
                tab.style.display = (index === tabIndex) ? 'block' : 'none';
            });
        };
        
        childWindow.cheatUpdateStat = function(key) {
            var value = childWindow.document.getElementById(key);
            if (!value) return;
            
            var inputValue = value.textContent.trim();
            var inputType = value.getAttribute('data-type');
            var errorValue = childWindow.document.getElementById(key + 'Error');
            
            if (inputType === 'number') {
                var numValue = parseInt(inputValue);
                if (!isNaN(numValue)) {
                    value.textContent = Math.round(numValue).toString();
                    if (errorValue) errorValue.style.display = 'none';
                    window.stats[key] = numValue;
                } else {
                    if (errorValue) {
                        errorValue.textContent = 'Please enter a valid number';
                        errorValue.style.display = 'inline-block';
                    }
                }
            } else {
                window.stats[key] = inputValue;
                if (errorValue) errorValue.style.display = 'none';
            }
        };
        
        childWindow.cheatModifyBoolean = function(key) {
            var checkbox = childWindow.document.getElementById(key);
            window.stats[key] = checkbox.checked;
        };
        
        childWindow.cheatModifyString = function(key) {
            var input = childWindow.document.getElementById(key);
            window.stats[key] = input.value;
        };
        
        childWindow.cheatFilterTable = function(searchId, tableId) {
            var filterValue = childWindow.document.getElementById(searchId).value.toLowerCase();
            var table = childWindow.document.getElementById(tableId);
            if (!table) return;
            
            var rows = table.getElementsByTagName('tr');
            for (var i = 0; i < rows.length; i++) {
                var cells = rows[i].getElementsByTagName('td');
                var rowVisible = false;
                for (var j = 0; j < cells.length; j++) {
                    var cellValue = cells[j].textContent || cells[j].innerText;
                    if (cellValue.toLowerCase().indexOf(filterValue) > -1) {
                        rowVisible = true;
                        break;
                    }
                }
                rows[i].style.display = rowVisible ? '' : 'none';
            }
        };
        
        childWindow.cheatSingleDrag = function(event, key) {
            var statBar = event.target;
            var dragStartX = 0;
            var dragStartValue = parseInt(statBar.style.width);
            var value = childWindow.document.getElementById(key);
            
            function startDrag(e) {
                dragStartX = e.clientX;
                dragStartValue = parseInt(statBar.style.width);
                childWindow.document.addEventListener('mousemove', doDrag);
                childWindow.document.addEventListener('mouseup', stopDrag);
            }
            
            function doDrag(e) {
                var dragDistance = e.clientX - dragStartX;
                var newValue = dragStartValue + (dragDistance / statBar.parentNode.clientWidth) * 100;
                newValue = Math.min(100, Math.max(1, newValue));
                statBar.style.width = newValue + '%';
                value.textContent = newValue.toFixed(0);
            }
            
            function stopDrag(e) {
                var newValue = parseInt(value.textContent);
                window.stats[key] = newValue;
                childWindow.document.removeEventListener('mousemove', doDrag);
                childWindow.document.removeEventListener('mouseup', stopDrag);
            }
            
            startDrag(event);
        };
        
        childWindow.cheatOpposedDrag = function(event, key, opposedKey) {
            var statBar = event.target;
            var dragStartX = 0;
            var dragStartValue = parseInt(statBar.style.width);
            var value = childWindow.document.getElementById(key);
            var opposedValue = childWindow.document.getElementById(opposedKey);
            
            function startDrag(e) {
                dragStartX = e.clientX;
                dragStartValue = parseInt(statBar.style.width);
                childWindow.document.addEventListener('mousemove', doDrag);
                childWindow.document.addEventListener('mouseup', stopDrag);
            }
            
            function doDrag(e) {
                var dragDistance = e.clientX - dragStartX;
                var newValue = dragStartValue + (dragDistance / statBar.parentNode.clientWidth) * 100;
                newValue = Math.min(100, Math.max(1, newValue));
                statBar.style.width = newValue + '%';
                value.textContent = newValue.toFixed(0);
                opposedValue.textContent = (100 - newValue).toFixed(0);
            }
            
            function stopDrag(e) {
                var newValue = parseInt(value.textContent);
                window.stats[key] = newValue;
                childWindow.document.removeEventListener('mousemove', doDrag);
                childWindow.document.removeEventListener('mouseup', stopDrag);
            }
            
            startDrag(event);
        };
        
        // Show first tab by default
        childWindow.cheatShowTab(0);
        
        console.log('Cheat window opened!');
    }
